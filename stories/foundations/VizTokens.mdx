import { getVizScaleTokens, getVizSizeTokens, getVizMarginToken } from '@/viz/tokens/scale-token-mapper.js';

const sequentialTokens = getVizScaleTokens('sequential');
const divergingTokens = getVizScaleTokens('diverging');

  borderBottom: '1px solid color-mix(in srgb, var(--cmp-border-default, #CBD5F5) 35%, transparent)',

const TokenTable = ({ tokens, title, renderPreview }) => (
  <section>
    <h3 style={{ fontSize: '1rem', marginBottom: '0.25rem' }}>{title}</h3>

      </thead>
      <tbody>
        {tokens.map((token) => (
          <tr key={typeof token === 'string' ? token : token.token}>

              <code>{typeof token === 'string' ? token : token.token}</code>
            </td>
            <td style={cellStyle}>{renderPreview(token)}</td>
          </tr>

# Viz Tokens

Tokenization for Sprint 21 visualizations combines governed OKLCH ramps with
existing system aliases so chart traits never reach directly into raw palette
files. Use these helpers to wire Vega-Lite configs, Tailwind themes or future
`--cmp-viz-*` slots.
